CREATE DATABASE inventariohw;
USE inventariohw;

CREATE TABLE utente(
id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(255),
cognome VARCHAR(255),
email VARCHAR(255),
password VARCHAR(64),
is_admin BOOLEAN);

CREATE TABLE aula(
id INT PRIMARY KEY AUTO_INCREMENT,
descrizione VARCHAR(255));

CREATE TABLE tipo(
id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(255),
descrizione VARCHAR(255));

CREATE TABLE componente(
id INT PRIMARY KEY AUTO_INCREMENT,
marca VARCHAR(255),
descrizione VARCHAR(255),
numero_seriale VARCHAR(255),
data_installazione DATE,
utente_id INT,
tipo_id INT,
aula_id INT,
FOREIGN KEY (utente_id) REFERENCES utente(id),
FOREIGN KEY (tipo_id) REFERENCES tipo(id),
FOREIGN KEY (aula_id) REFERENCES aula(id));
